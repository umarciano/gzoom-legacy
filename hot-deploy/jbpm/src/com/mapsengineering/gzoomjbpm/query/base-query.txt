select execution.dbid_ as execution_id, execution.id_ as process_id, deplprop.objname_ as process_name, variable.key_ as var_name, variable.string_value_ as string_value
from jbpm4_deployprop as deplprop, 
     jbpm4_execution as execution, 
     jbpm4_variable as variable 
where deplprop.key_ = 'pdid'
      and deplprop.stringval_ = execution.procdefid_
      and execution.dbid_ = variable.execution_
order by variable.execution_
